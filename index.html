-- 1) 테이블 생성 (처음 설정 시)
create table if not exists language_sentences (
  id          text primary key,
  type        text, subtype text, category text, situation text,
  uzbek       text, korean text, english text, ipa text, note text,
  created_at  timestamptz default now(), updated_at timestamptz default now(), model text,
  audio_url   text, learning_json text
);
alter table language_sentences enable row level security;
drop policy if exists "allow_all" on language_sentences;
create policy "allow_all" on language_sentences
  for all using (true) with check (true);

-- 2) 동기화 메타 테이블 (처음 설정 시)
create table if not exists language_sync_meta (
  key text primary key,
  value text,
  updated_at timestamptz default now()
);
alter table language_sync_meta enable row level security;
drop policy if exists "allow_all_language_sync_meta" on language_sync_meta;
create policy "allow_all_language_sync_meta" on language_sync_meta
  for all using (true) with check (true);

-- 3) 버전별 스냅숏 테이블
create table if not exists language_sentence_versions (
  canonical_version text not null,
  record_id text not null,
  type text, subtype text, category text, situation text,
  uzbek text, korean text, english text, ipa text, note text,
  created_at timestamptz default now(), updated_at timestamptz default now(), model text,
  audio_url text, learning_json text,
  primary key (canonical_version, record_id)
);
alter table language_sentence_versions enable row level security;
drop policy if exists "allow_all_language_sentence_versions" on language_sentence_versions;
create policy "allow_all_language_sentence_versions" on language_sentence_versions
  for all using (true) with check (true);

-- 4) revision 비교 후 새 스냅숏을 원자적으로 활성화하는 함수
create or replace function language_activate_snapshot(
  p_expected_revision bigint,
  p_new_version text,
  p_record_count integer,
  p_force boolean default false
) returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  current_value jsonb;
  current_revision bigint;
  actual_count integer;
begin
  insert into language_sync_meta(key, value, updated_at)
  values ('state', '{"revision":0,"canonical_version":"legacy"}', now())
  on conflict (key) do nothing;

  select value::jsonb into current_value
  from language_sync_meta
  where key = 'state'
  for update;

  current_revision := coalesce((current_value->>'revision')::bigint, 0);
  if not p_force and current_revision <> p_expected_revision then
    return jsonb_build_object(
      'ok', false,
      'conflict', true,
      'revision', current_revision,
      'canonical_version', coalesce(current_value->>'canonical_version', 'legacy')
    );
  end if;

  select count(*) into actual_count
  from language_sentence_versions
  where canonical_version = p_new_version;

  if actual_count <> p_record_count then
    raise exception 'snapshot count mismatch: expected %, actual %', p_record_count, actual_count;
  end if;

  current_revision := current_revision + 1;
  update language_sync_meta
  set value = jsonb_build_object(
        'revision', current_revision,
        'canonical_version', p_new_version
      )::text,
      updated_at = now()
  where key = 'state';

  return jsonb_build_object(
    'ok', true,
    'conflict', false,
    'revision', current_revision,
    'canonical_version', p_new_version
  );
end;
$$;
grant execute on function language_activate_snapshot(bigint, text, integer, boolean) to anon, authenticated;
